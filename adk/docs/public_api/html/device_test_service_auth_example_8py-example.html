device_test_service_auth_example.py

This is an example using Python 2.7.

The Pycryptoplus library is used because the more common pycrypto library does not support AES-CMAC. Python 3 has native support for AES-CMAC, but requires the keys in a different format.

9 import codecs # Use codecs library for hex encode and decode
10 from CryptoPlus.Cipher import python_AES as AES
11 
12 # The key can be defined as an array
13 key = [0x7b, 0xe, 0x19, 0xc6, 0xd9, 0xb7, 0x4a, 0xcc, 0x99, 0x6d, 0x35, 0x61, 0xa9, 0x74, 0x5a, 0x7f]
14 # Or alternatively as a string, but that needs decoding before initialising the CMAC
15 key_string = "7b0e19c6d9b74acc996d3561a9745a7f"
16 print("Key:",key_string)
17 key = codecs.decode(key_string,'hex')
18 
19 # Initialise the encoder
20 aes = AES.new(key, AES.MODE_CMAC)
21 
22 challenge = "+AUTHSTART:B1978E453D8851D88D718C82953FF627"
23 print("Received challenge: ",challenge)
24 
25 # Extract the code from authstart
26 match = re.match("\+AUTHSTART\s*:\s*([0-9A-Fa-f]{32})", challenge)
27 random = codecs.decode(match.group(1), 'hex')
28 
29 # Encrypt the random number to generate a response
30 enc = aes.encrypt(random)
31 
32 # Then encode to a string, suitable for the response
33 enc_str = codecs.encode(enc, "hex")
34 response = "AT+AUTHRESP={}".format(enc_str.upper())
35 print("Response:",response)
